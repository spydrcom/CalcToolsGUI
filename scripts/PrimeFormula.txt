!! mellin (k) = ( (k - 1)! + 1 ) / k

CONFIGURE TRIG OP "cos" POW "2" description "cos squared" ; INSTANCE cosSq TRIG

!! count (i) = SIGMA [ 1 <= j <= i ] ( floor ( cosSq ( pi * mellin (j) ) ) )

!! find (n) = 1 + SIGMA [ 1 <= i <= 2^n ] ( floor ( ( n / count (i) )^(1/n) ) )
