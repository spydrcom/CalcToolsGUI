instance BIp3 Bessel

instance BIn3 Bessel

calc getBk3Alpha()

prettyprint kalpha

// Airy Ai/Bi functions
!! Ai(x) = 1/pi * sqrt(x/3) * BK3( 2/3 * sqrt(x^3) )
!! Bi(x) = sqrt(x/3) * ( BIp3( 2/3 * sqrt(x^3) ) + BIn3( 2/3 * sqrt(x^3) ) )


// function renders

RENDERF Ai
RENDERF Bi
